#include <iostream>
#include <string>

class EmailService {
public:
    void sendEmail(std::string recipient, std::string message) {
        std::cout << "Sending an email to " << recipient << " with message: " << message << std::endl;
    }
};

class NotificationService {
private:
    EmailService emailService;

public:
    NotificationService(EmailService service) : emailService(service) {}

    void notifyUser(std::string recipient, std::string message) {
        emailService.sendEmail(recipient, message);
    }
};

int main() {
    EmailService emailService;
    NotificationService notificationService(emailService);

    notificationService.notifyUser("example@example.com", "Hello! This is a notification.");

    return 0;
}
